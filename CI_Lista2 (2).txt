cod_peca integer,
nome character varying,
cor character varying,
Peso float,
cidade character varying,
primary key(cod_peca)
);

create table lista2.embarque(
cod_peca integer,
cod_fornec integer,
quantidade integer,
primary key(cod_peca,cod_fornec,quantidade),
foreign key(cod_peca) references lista2.pecas(cod_peca),
foreign key (cod_fornec) references lista2.fornecedor(cod_fornec)
);
